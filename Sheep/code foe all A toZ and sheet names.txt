Sub SortSheetsAlphabetically()								
    Dim i As Integer, j As Integer								
    Dim SheetCount As Integer								
								
    SheetCount = ThisWorkbook.Sheets.Count								
								
    For i = 1 To SheetCount - 1								
        For j = i + 1 To SheetCount								
            If UCase(ThisWorkbook.Sheets(j).Name) < UCase(ThisWorkbook.Sheets(i).Name) Then								
                ThisWorkbook.Sheets(j).Move Before:=ThisWorkbook.Sheets(i)								
            End If								
        Next j								
    Next i								
								
    MsgBox "Sheets have been sorted alphabetically.", vbInformation								
End Sub

Sub ListAllSheetNames()
    Dim ws As Worksheet
    Dim summarySheet As Worksheet
    Dim i As Long
    
    On Error Resume Next
    ' Try to delete the existing summary sheet if it exists
    Application.DisplayAlerts = False
    Sheets("SheetList").Delete
    Application.DisplayAlerts = True
    On Error GoTo 0
    
    ' Add a new sheet for the summary
    Set summarySheet = ThisWorkbook.Sheets.Add
    summarySheet.Name = "SheetList"
    
    ' Loop through all sheets and write their names
    i = 1
    For Each ws In ThisWorkbook.Sheets
        If ws.Name <> summarySheet.Name Then
            summarySheet.Cells(i, 1).Value = ws.Name
            i = i + 1
        End If
    Next ws
    
    MsgBox "All sheet names listed in 'SheetList'.", vbInformation
End Sub

    function addSheep() {
      const name = document.getElementById('sheepName').value.trim();
      const age = document.getElementById('sheepAge').value.trim();
      const breed = document.getElementById('sheepBreed').value.trim();
      const health = document.getElementById('sheepHealth').value;

      if (!name || !age || !breed || !health) {
        alert("Please fill out all fields.");
        return;
      }

      const newSheepRef = sheepRef.push();
      newSheepRef.set({
        name,
        age: Number(age),
        breed,
        health,
        createdAt: Date.now()
      });

      // Clear inputs
      document.getElementById('sheepName').value = '';
      document.getElementById('sheepAge').value = '';
      document.getElementById('sheepBreed').value = '';
      document.getElementById('sheepHealth').selectedIndex = 0;
    }

    function removeSheep(key) {
      sheepRef.child(key).remove();
    }

    sheepRef.on('value', snapshot => {
      const sheepData = snapshot.val();
      const sheepList = document.getElementById('sheepList');
      sheepList.innerHTML = '';

      if (!sheepData) {
        sheepList.innerHTML = '<p>No sheep records found.</p>';
        return;
      }

      Object.entries(sheepData).forEach(([key, sheep]) => {
        const div = document.createElement('div');
        div.className = 'sheep';
        div.innerHTML = `
          <div>
            <strong>${sheep.name}</strong> — Age: ${sheep.age}, Breed: ${sheep.breed}, Health: <em>${sheep.health}</em>
          </div>
          <button onclick="removeSheep('${key}')">Remove</button>
        `;
        sheepList.appendChild(div);
      });
    });
